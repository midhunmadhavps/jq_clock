//CSS PART body{ text-align: center; background-color:#608c8f; padding:20%; } .clock{ background-color:#2e4c4e; padding:10%; box-shadow: -1px 0px 30px -2px rgb(126,200,204); font-size:40px; color: #fff; } //JS PART $(document).ready(function(){ showtime(); function showtime(){ var date = new Date(); var h = date.getHours(); var m = date.getMinutes(); var s = date.getSeconds(); var session = "AM"; if (h >= 12){ session = "PM" } if( h == 0){ h=12; } if(h>12){ h = h - 12; } if(h < 10){ h = "0" + h; } if(m < 10){ m = "0" + m; } if(s < 10){ s = "0" + s; } var time = h + ":" + m + ":" + s + " " + session; $("#clock").html(time); setTimeout(showtime,1000); } });
